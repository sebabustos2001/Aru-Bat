# Generar servicio para ejecucion automatica al encender RaspberryPi5
[Unit]
Description=Ejecutar script __main__.py despues de arranque OS
After=network.target

[Service]
#Type=simple
ExecStart=/usr/bin/python3 /home/admin/ARU/ARU_BAT/__main__.py
WorkingDirectory=/home/admin/ARU/ARU_BAT
StandardOutput=inherit
StandardError=inherit
Restart=always
User=admin  -> Cambiar por el nombre de usuario en el OS

[Install]
WantedBy=multi-user.target
